# 闭包

# 理解闭包

  • 是什么？
    • 闭包就是保护一个可重用的局部变量的词法结构
  • 为什么？
    • 全局变量可重用，但易污染；局部变量不可重用；当我们想创建一个可重用的局部变量时，就需要闭包
  • 怎么做？
    • 外层函数封装受保护变量和专门操作变量的内层函数
    • 外层函数将操作变量的内层函数返回到外部
    • 调用外层函数即获得内层函数的对象

# 排序

sort可以完成单位数字排序，但是多位数字如10和2，会出现2比10大的情况，可以使用一个比较函数来避免此类情况发生。

// 简单的排序函数 无法正确比较多位数字
const arr=[10,9,8]
arr.sort() //[ 10, 8, 9 ]

// 使用一个compare函数
const simpleCompare = (value1, value2) => {
    return value1 - value2;
}
arr.sort(simpleCompare) //[ 8, 9, 10 ]

// 可以比较数组中的某个对象中的任一数据
const compare=(params)=>{
    return function(value1,value2){
        return value1[params]-value2[params]
    }
}

# 随机打乱数组顺序的方法

// 方法1：利用随机数和排序，缺点有一定随机性，但分布不均匀
const arr = [1, 2, 3, 4, 5];
let randomArr = array => {
    array.sort(() => Math.random() - 0.5);
    return array;
}
randomArr(arr);

// 方法2：循环随机交换位置
const arr = [1, 2, 3, 4, 5];
let randomArr = arr => {
    for (let i = 0; i < arr.length; i++) {
        let index = parseInt(Math.random() * (arr.length - 1));
        arr[i] = [arr[index], arr[index] = arr[i]][0];
        return arr;
    }
}
randomArr(arr);

# 浅谈递归

简单来说就是函数自己调用自己。

  • 需要有明确的函数出口条件
// 递归处理年龄问题
function recursion(num) {
    if (num > 1) {
        return 2 + arguments.callee(num - 1);
        // return 2 + recursion(num - 1)(num - 1);
    } else {
        return 10
    }
}
recursion(8) //24

const paramQuery = window.location.search.substring(1);

let check = value => {
    for (let i=0; i < params.length; i++) {

# 统计字符串中字符出现次数

let str = "12rjh139yug9hf1ohfufhwafoglh1"

// 方法1：
let str = "12rjh139yug9hf1ohfufhwafoglh1"
let obj={}
for(let i=0;i<str.length;i++){
    // 如果已存在则加1
    if(obj[str[i]]){
        obj[str[i]]++
    }
    // 不存在则初始化
    else{
        obj[str[i]]=1
    }
}
console.log(obj)

// ------------------------------------
// 方法2：
let arr = str.split("")
// 冒泡排序
for (let i = 0; i < arr.length; i++) {
    for (let j = 0; j < arr.length; j++) {
        if (arr[j - 1] > arr[j]) {
            arr[j] = [arr[j - 1], arr[j - 1] = arr[j]][0]
        }
    }
}
let count = 1;
let obj = {};
// 统计次数
for (let i = 1; i <= arr.length; i++) {
    if (arr[i - 1] == arr[i]) {
        count++
    } else {
        obj[arr[i - 1]] = count;
        count = 1;
    }
}

# 数据类型问题

# typeof 和 instanceof

  • typeof:简单的类型识别方法，仅用于识别基本数据类型，引用数据类型(Array,null)时，都会返回object
  • instanceof:基于原型链的识别方式，判断原型链中是否存在一个构造函数的prototype属性，用于复杂数据类型
console.log(typeof null);  //object
console.log(typeof NaN);   //number
console.log(0.1+0.2===0.3); //false
console.log(2&&'a') //a
console.log('a'&&2) //2
console.log(2 || 'a') //2
console.log('5'<'10') //false
console.log('5'>'10') //true
console.log('23'>'3') //false

# 类型转换

# 阿拉伯数字转中文

function numToString(value){
    let number=value.split('')
    const numberCN=["零","一","二","三","四","五","六","七","八","九"]
    const unit=["","十","百","千","万"]
    number.forEach((item,index,arr)=>{
        arr[index]=numberCN[item]
    })
    number.reverse()
    function recursion(num){
        if(num<5&&num>0){
            return number[num]+unit[num]+recursion(num-1)
        }
        else if(num>=5){
            return number[num]+recursion(num-1)
        }
        else{
            return number[num]
        }
    }
    number=recursion(number.length-1)
    return number
}
console.log(numToString("12345"))  //一万二千三百四十五

// 1)根据分数排序

// --------------方法1：----------------
function compare(property){
    return function(a,b){
        let value1=a[property];
        let value2=b[property];
        return value1-value2
    }
}
// 备份数据
let sortArr=[...arr]
sortArr.sort(compare('score'))
console.log(sortArr);
// 反序
sortArr.reverse()
console.log(sortArr);

// --------------方法2：--------------

// 备份数据
let cloneArr=[...arr]
// 从小到大冒泡排序
for(let i=0;i<cloneArr.length;i++){
    for(let i=1;i<cloneArr.length;i++){
        if(cloneArr[i-1].score>cloneArr[i].score){
            cloneArr[i-1]=[cloneArr[i],cloneArr[i]=cloneArr[i-1]][0]

    }
}
